async function analyzeImage(type) { const apiKey = db.settings.apiKey; if(!apiKey) return alert('請先到設定頁面填入 Gemini API Key'); const fileInput = document.getElementById(type === 'diet' ? 'diet-image' : 'ex-image'); const file = fileInput.files[0]; if(!file) return; const base64Image = await toBase64(file); console.log("正在發送請求至 Gemini..."); try { // 使用更穩定的 v1beta 介面與 gemini-1.5-flash-latest 模型 const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [ { text: type === 'diet' ? "你是一位專業的營養師。請分析這張食物照片，估算其名稱與總熱量，並嚴格以 JSON 格式回傳，不要有任何額外文字：{ \"name\": \"食物名稱\", \"calories\": 數字 }" : "請分析這張運動紀錄截圖，提取運動類型與消耗熱量，並嚴格以 JSON 格式回傳：{ \"name\": \"運動類型\", \"calories\": 數字 }" }, { inline_data: { mime_type: file.type, data: base64Image.split(',')[1] } } ] }] }) }); const data = await response.json(); // 檢查 API 是否回傳錯誤 if (data.error) { throw new Error(data.error.message); } if (data.candidates && data.candidates[0].content && data.candidates[0].content.parts) { const text = data.candidates[0].content.parts[0].text; // 過濾掉 Markdown 格式標籤 (如 ```json) const cleanJson = text.replace(/```json|```/g, "").trim(); const result = JSON.parse(cleanJson); db[type].push({ name: result.name || "未知項目", calories: parseInt(result.calories) || 0, date: new Date().toLocaleDateString(), id: Date.now() }); saveDB(); type === 'diet' ? renderDiet() : renderExercise(); renderDashboard(); alert('辨識成功！'); } else { throw new Error("API 回傳格式不正確"); } } catch (e) { console.error("詳細錯誤訊息:", e); alert('辨識失敗：' + e.message + '\n請確認 API Key 是否擁有 Gemini 1.5 Flash 的權限。'); } }
